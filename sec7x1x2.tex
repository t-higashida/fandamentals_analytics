\subsection{積分の性質}

つぎの定理は1変数の関数の定積分の性質に関する定理4.1の類似である．

\定理{7.1}
$ f(P) = f(x,y) , g(P) = g(x,y), P=(x,y)$，
を矩形$ K= \{ (x,y)  \: | \:  a \leqq x \leqq b , c \leqq y \leqq d \}$において連続な関数とナる．
\begin{list}{}{}
	\item	$(1^\circ)$ $a = \alpha_0 < \alpha_1 < \cdots < \alpha_p = b , c= \gamma_1 < \cdots < \gamma_i < \cdots < \gamma_q = d$として
$K$を$pq$個の矩形$K_{hi} = \{ (x,y) \: | \: \alpha_{h-1} \leqq x \leqq \alpha_h , \: \gamma_{i-1} \leqq y \leqq \gamma_i \}, \: h=1,2, \cdots , p , i=1,2, \cdots , q$，
に分割すれば
\begin{equation}
	\int_K f(x,Y dx dy = \sum_{h=1}^{p}	\sum_{i=1}^{q}	\int_{K_{hi}} f(x,y) dx dy.
\label{sec7x5}
\end{equation}
	\item $(2^\circ)$ $c_1 , c_2$を任意の定数としたとき
\begin{equation}
	\int_K ( c_1 f(P) + c_2 g(P) ) dw = c_1 \int_K f(P) dw + c_2 \int_K g(P) dw.
\label{sec7x6}
\end{equation}
	\item $(3^\circ)$
矩形$K$においてつねに$f(x,y) \geqq g(x,y)$ならば
\[
	\int_K f(x,y) dx dy \geqq \int_K g(x,y) dx dy
\]
であって，$K$で恒等的に$f(x,y) = g(x,y)$である場合を除けば
\[
	\int_K f(x,y) dx dy > \int_K g(x,y) dx dy.
\]
	\item $(4^\circ)$
\begin{equation}
	\left|
		\int_K f(x,y ) dx dy
	\right|
	\leqq	\int_K | f(x,y) | dx dy.
\label{sec7x7}
\end{equation}
\end{list}

\証明

$(1^\circ)$
積分の定義7.1 において分割$\varDelta$を
$ \alpha_h \in \{ x_0 , x_1 , \cdots , x_m \}, \gamma_i \in \{ y_0 , y_1 , \cdots , y_n \}$
なるように選んで $ \alpha_h = x_j(h) , \gamma_k(i)$とすれば，（\ref{eq7x4}）の記法を用いて
\[
	\sum_{j=1}^{m}
	\sum_{k=1}^{n}	f(P_{jk}) \omega_{jk}
	=
	\sum_{h=1}^{p}
	\sum_{i=1}^{q}
	\sum_{j=j(h-1)+1}^{j(h)}
	\sum_{k=k(i-1)+1}^{k(i)}
	f(P_{jk}) \omega_{jk}.
\]
この両辺の$ \delta [ \varDelta ] \to 0 $のときの極限をとれば直ちに
（\ref{eq7x5}）を得る．
$(2^\circ)$，$(3^\circ)$，$(4^\circ)$は定理4.1，$(2^\circ)$，$(3^\circ)$，$(4^\circ)$，$(5^\circ)$と同様にして証明される．
